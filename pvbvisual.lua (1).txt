-- Function when button is clicked
local function onButtonClicked()
    local player = game.Players.LocalPlayer
    local tool = player.Character and player.Character:FindFirstChildOfClass("Tool")

    if tool then
        -- Visual feedback
        dupeButton.BackgroundColor3 = Color3.fromRGB(40, 120, 80)
        dupeButton.Text = "✓ Duplicated Successfully!"
        statusLabel.Text = "Status: Duplicated - " .. tool.Name
        statusLabel.TextColor3 = Color3.fromRGB(100, 255, 150)
        
        -- Duplicate the item
        local clone = tool:Clone()
        clone.Parent = player.Backpack
        
        -- Reset button after delay
        wait(2)
        dupeButton.BackgroundColor3 = Color3.fromRGB(35, 40, 45)
        dupeButton.Text = "Start Duplication"
        statusLabel.Text = "Status: Ready"
        statusLabel.TextColor3 = Color3.fromRGB(100, 200, 255)
    else
        -- No tool equipped feedback
        dupeButton.BackgroundColor3 = Color3.fromRGB(120, 40, 40)
        dupeButton.Text = "✗ No Item Equipped!"
        statusLabel.Text = "Status: Error - Equip an item first"
        statusLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
        
        wait(2)
        dupeButton.BackgroundColor3 = Color3.fromRGB(35, 40, 45)
        dupeButton.Text = "Start Duplication"
        statusLabel.Text = "Status: Ready"
        statusLabel.TextColor3 = Color3.fromRGB(100, 200, 255)
    end
end

-- Connect button click
dupeButton.MouseButton1Click:Connect(onButtonClicked)

-- Make frame draggable
local dragging
local dragInput
local dragStart
local startPos

local function update(input)
    local delta = input.Position - dragStart
    backgroundFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

backgroundFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = backgroundFrame.Position
        
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

backgroundFrame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        dragInput = input
    end
end)

game:GetService("UserInputService").InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        update(input)
    end
end)
